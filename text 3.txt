    let currentSection = 0;
    let currentQuestion = 0;
    let selectedAnswer = null;
    let showResult = false;
    let score = 0;
    let answeredQuestions = 0;

    const totalQuestions = sections.reduce((sum, section) => sum + section.questions.length, 0);

    function render() {
        const container = document.getElementById('quiz-container');
        
        if (answeredQuestions === totalQuestions) {
            container.innerHTML = renderResults();
        } else {
            container.innerHTML = renderQuestion();
        }
        
        attachEventListeners();
    }

    function renderQuestion() {
        const currentQ = sections[currentSection].questions[currentQuestion];
        const progress = (answeredQuestions / totalQuestions) * 100;
        
        return `
            <div class="header">
                <h1>Financial Data Sources Quiz</h1>
                <div class="section-info">
                    <div class="section-title">Section ${currentSection + 1}: ${sections[currentSection].title}</div>
                    <div class="score">Score: ${score}/${answeredQuestions}</div>
                </div>
                <div class="progress-bar">
                    <div class="progress-fill" style="width: ${progress}%"></div>
                </div>
                <div class="progress-text">Question ${answeredQuestions + 1} of ${totalQuestions}</div>
            </div>
            
            <div class="question-card">
                <div class="question-badge">
                    Question ${currentQuestion + 1} of ${sections[currentSection].questions.length}
                </div>
                
                <div class="question-text">${currentQ.question}</div>
                
                <div class="options">
                    ${currentQ.options.map((option, index) => {
                        let className = 'option';
                        if (showResult) {
                            className += ' disabled';
                            if (index === currentQ.correct) className += ' correct';
                            if (index === selectedAnswer && index !== currentQ.correct) className += ' wrong';
                        }
                        return `<div class="${className}" data-index="${index}">${option}</div>`;
                    }).join('')}
                </div>
                
                ${showResult ? `
                    <div class="feedback ${selectedAnswer === currentQ.correct ? 'correct' : 'wrong'}">
                        ${selectedAnswer === currentQ.correct 
                            ? '‚úì Correct! Well done!' 
                            : '‚úó Incorrect. The correct answer is highlighted above.'}
                    </div>
                    <button class="next-btn" id="next-btn">
                        ${currentQuestion < sections[currentSection].questions.length - 1 || currentSection < sections.length - 1
                            ? 'Next Question ‚Üí'
                            : 'See Results ‚Üí'}
                    </button>
                ` : `
                    <div class="hint">üí° Select an answer above to continue</div>
                `}
            </div>
        `;
    }

    function renderResults() {
        const percentage = Math.round((score / totalQuestions) * 100);
        let message = '';
        
        if (percentage >= 90) message = "Outstanding! You're a financial data expert! üåü";
        else if (percentage >= 75) message = "Excellent work! You've got a solid grasp! üëè";
        else if (percentage >= 60) message = "Good job! Keep learning! üìö";
        else if (percentage >= 50) message = "Not bad! Review the material and try again! üí™";
        else message = "Keep practicing! You'll get there! üéØ";
        
        return `
            <div class="results-screen">
                <div class="results-card">
                    <div class="trophy">üèÜ</div>
                    <div class="results-title">Quiz Complete!</div>
                    <div class="final-score">${score}/${totalQuestions}</div>
                    <div class="percentage">${percentage}%</div>
                    <div class="message">${message}</div>
                    <button class="restart-btn" id="restart-btn">üîÑ Try Again</button>
                </div>
            </div>
        `;
    }

    function attachEventListeners() {
        if (!showResult) {
            document.querySelectorAll('.option:not(.disabled)').forEach(option => {
                option.addEventListener('click', function() {
                    handleAnswer(parseInt(this.dataset.index));
                });
            });
        }
        
        const nextBtn = document.getElementById('next-btn');
        if (nextBtn) {
            nextBtn.addEventListener('click', handleNext);
        }
        
        const restartBtn = document.getElementById('restart-btn');
        if (restartBtn) {
            restartBtn.addEventListener('click', restart);
        }
    }

    function handleAnswer(index) {
        if (showResult) return;
        
        selectedAnswer = index;
        showResult = true;
        answeredQuestions++;
        
        if (index === sections[currentSection].questions[currentQuestion].correct) {
            score++;
        }
        
        render();
    }

    function handleNext() {
        if (currentQuestion < sections[currentSection].questions.length - 1) {
            currentQuestion++;
        } else if (currentSection < sections.length - 1) {
            currentSection++;
            currentQuestion = 0;
        }
        
        selectedAnswer = null;
        showResult = false;
        render();
    }

    function restart() {
        currentSection = 0;
        currentQuestion = 0;
        selectedAnswer = null;
        showResult = false;
        score = 0;
        answeredQuestions = 0;
        render();
    }

    render();
</script>
